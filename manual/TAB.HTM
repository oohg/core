#command DEFINE TAB <name> ;
      [ OBJ <obj> ] ;
      [ <dummy1: OF, PARENT> <parent> ] ;
      AT <row>, <col> ;
      WIDTH <w> ;
      HEIGHT <h> ;
      [ VALUE <value> ] ;
      [ FONT <f> ] ;
      [ SIZE <s> ] ;
      [ <bold: BOLD> ] ;
      [ <italic: ITALIC> ] ;
      [ <underline: UNDERLINE> ] ;
      [ <strikeout: STRIKEOUT> ] ;
      [ TOOLTIP <tooltip> ] ;
      [ <buttons: BUTTONS> ] ;
      [ <flat: FLAT> ] ;
      [ <hottrack: HOTTRACK> ] ;
      [ <vertical: VERTICAL> ] ;
      [ <dummy03: ONCHANGE, ON CHANGE> <change> ] ;
      [ <notabstop: NOTABSTOP> ] ;
      [ <rtl: RTL> ] ;
      [ SUBCLASS <subclass> ] ;
      [ <internals: INTERNALS> ] ;
      [ <invisible: INVISIBLE> ] ;
      [ <disabled: DISABLED> ] ;
      [ <multiline: MULTILINE> ] ;
      [ <noproc: DELAYMSGS> ] ;
   => ;
      [ <obj> := ] _OOHG_SelectSubClass( TTab(), [ <subclass>() ] ): ;
            Define( <(name)>, <(parent)>, <col>, <row>, <w>, <h>, {}, {}, ;
            <value>, <f>, <s>, <tooltip>, <{change}>, <.buttons.>, <.flat.>, ;
            <.hottrack.>, <.vertical.>, <.notabstop.>,, <.bold.>, <.italic.>, ;
            <.underline.>, <.strikeout.>, {}, <.rtl.>, <.internals.>, ;
            <.invisible.>, <.disabled.>, <.multiline.>, <.noproc.> )

#command PAGE <caption> ;
      [ IMAGE <image> ] ;
      [ NAME <name> ] ;
      [ OBJ <obj> ] ;
      [ SUBCLASS <subclass> ] ;
   => ;
      [ <obj> := ] ;
            _BeginTabPage( <caption>, <image>,, <(name)>, [ <subclass>() ] )

#command DEFINE PAGE <caption> ;
      [ IMAGE <image> ] ;
      [ NAME <name> ] ;
      [ OBJ <obj> ] ;
      [ SUBCLASS <subclass> ] ;
   => ;
      [ <obj> := ] ;
            _BeginTabPage( <caption>, <image>,, <(name)>, [ <subclass>() ] )

#command DEFINE TAB PAGE <caption> ;
      [ IMAGE <image> ] ;
      [ NAME <name> ] ;
      [ OBJ <obj> ] ;
      [ SUBCLASS <subclass> ] ;
   => ;
      [ <obj> := ] ;
            _BeginTabPage( <caption>, <image>,, <(name)>, [ <subclass>() ] )

#command END PAGE ;
   => ;
      _EndTabPage()

#command END TAB ;
   => ;
      _EndTab()
