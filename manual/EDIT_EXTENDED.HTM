#xtranslate EDIT EXTENDED                              ;
                [ WORKAREA <cWorkArea> ]               ;
                [ TITLE <cTitle> ]                     ;
                [ FIELDNAMES <acFieldNames> ]          ;
                [ FIELDMESSAGES <acFieldMessages> ]    ;
                [ FIELDENABLED <alFieldView> ]         ;
                [ TABLEVIEW <alTableView> ]            ;
                [ OPTIONS <aOptions> ]                 ;
                [ ON SAVE <bSave> ]                    ;
                [ ON FIND <bFind> ]                    ;
                [ ON PRINT <bPrint> ]                  ;
        =>                         ;
                ABM2(   <(cWorkArea)>,                  ;
            <cTitle>,                 ;
            <acFieldNames>,             ;
            <acFieldMessages>,                 ;
            <alFieldView>,             ;
            <alTableView>,                      ;
                        <aOptions>,                    ;
                        <bSave>,                          ;
                        <bFind>,                          ;
                        <bPrint> )
