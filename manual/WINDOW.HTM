    =>;
        DoMethod ( <(name)> , 'Release' )

    =>;
    ReleaseAllWindows()

    =>;
    ReleaseAllWindows()

#xtranslate EXIT PROGRAM  ;
    =>;
    ReleaseAllWindows()
